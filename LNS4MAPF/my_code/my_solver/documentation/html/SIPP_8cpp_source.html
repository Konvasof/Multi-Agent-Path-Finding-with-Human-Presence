41 auto SIPPNodeComparator::operator()(SIPPNode const* n1, SIPPNode const* n2) const -> bool
42 {
43  // compare f
44  if (n1->f == n2->f)
45  {
46  if (n1->h == n2->h)
47  { // tie in reach time
48  if (n1->h2 == n2->h2)
49  { // tie in distance from goal
50  if (n1->h3 == n2->h3)
51  {
52  // if tie in all heuristics, choose randomly
53  return dist(*rnd_generator);
54  }
55  // if f is the same and h is the same and h2 is the same prefer locations that dont block goals of other agents
56  return n1->h3 > n2->h3;
57  }
58  return n1->h2 > n2->h2; // if f is the same and h is the same prefer nodes closer to goal
59  }
60  return n1->h > n2->h; // if f is the same, prefer earlier intervals
61  }
62  return n1->f > n2->f;
63 }

65 auto SIPPNodeComparatorSuboptimal::operator()(SIPPNode const* n1, SIPPNode const* n2) const -> bool
66 {
67  // compare f
68  if (std::abs(n1->f - n2->f) <= max_suboptimality_absolute)
69  {
70  if (n1->h == n2->h)
71  { // tie
72  if (n1->h2 == n2->h2)
73  {
74  if (n1->f == n2->f)
75  {
76  if (n1->h3 == n2->h3)
77  {
78  // if tie, choose randomly
79  return dist(*rnd_generator);
80  }
81  return n1->h3 > n2->h3;
82  }
83  return n1->f > n2->f;
84  }
85  return n1->h2 > n2->h2;
86  }
87  return n1->h > n2->h;
88  }
89  return n1->f > n2->f;
90 }

100 auto NodePool::add_node(SIPPNode node) -> SIPPNode*
101 {
102  // check, whether there is still space in the node pool
103  if (end < (int)pool.size())
104  {
105  // move to the end of the pool and increment the count
106  pool[end++] = std::move(node);
107  return &pool[end - 1];
108  }
109 
110  // if there is no space in the vector, add to the list, which will be merged with the vector at the end of the iteration
111  extra.push_back(node);
112  return &extra.back();
113 }

116 void NodePool::merge_extra()
117 {
118  if (extra.size() > NODE_POOL_MERGE_SIZE)
119  {
120  // merge the list with the vector
121  // be careful, this invalidates all pointers to the nodes in the node pool
122  pool.insert(pool.end(), std::move_iterator{extra.begin()}, std::move_iterator{extra.end()});
123  extra.clear();
124  }
125  // reset the node counter
126  end = 0;

944 auto sipp::extract_path(SIPPNode const* const final_node) -> TimePointPath
945 {
946  TimePointPath path;
947  SIPPNode const* curr = final_node;
948  SIPPNode const* prev = nullptr;
949  while (curr != nullptr)
950  {
951  if (prev == nullptr) // push the whole last interval
952  {
953  path.push_back(curr->time_point);
954  }
955  else // push only the occupied part of other intervals
956  {
957  path.push_back(
958  TimePoint(curr->time_point.location, TimeInterval(curr->time_point.interval.t_min, prev->time_point.interval.t_min - 1)));
959  }
960  prev = curr;
961  curr = curr->parent;
962  }
963  std::reverse(path.begin(), path.end());
964  return path;
