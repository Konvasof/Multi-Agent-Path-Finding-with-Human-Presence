33 void Instance::initialize(bool calculate_manhattan, bool calculate_euclidean)
34 {
35  assertm(!initialized, "Instance already initialized.");
36  load_map();
37  load_scene();
38  precompute_neighbors();
39  calculate_heuristics(calculate_manhattan, calculate_euclidean);
40  // calculate sum of distances
41  sum_of_distances = 0;
42  for (int i = 0; i < num_of_agents; i++)
43  {
44  sum_of_distances += get_heuristic_distance(i, start_locations[i]);

46

51  // for each location remember the number of agent whose goal it is
52  location_to_goal_array[goal_locations[i]] = i;

96 void Instance::load_scene()
97 {
98  // open the scene file
99  std::ifstream scene_file(scene_fname.c_str());
100 
101  // check if the open was successful
102  if (!scene_file.is_open())
103  {
104  throw std::runtime_error("Unable to open scene file.");
105  }
106 
107  std::string line;
108 
109  // read the version
110  if (!getline(scene_file, line) || line != "version 1")
111  {
112  throw std::invalid_argument("Invalid version of scene file.");
113  }
114 
115  int loaded_agents = 0;
116  start_positions.reserve(num_of_agents);
117  goal_positions.reserve(num_of_agents);
118  // read the file line by line until enough agents were loaded
119  while (loaded_agents < num_of_agents && getline(scene_file, line))
120  {
121  // split the line - the format is 'bucket, map, map_width, map_height, start_x, start_y, goal_x, goal_y, optimal_length'
122  std::vector<std::string> split_line;
123  boost::split(split_line, line, boost::is_any_of("\t "));
124 
125  // filter out invalid lines
126  if (split_line.size() != 9)
127  {
128  std::cout << "Invalid line in scen file: " << line << " length: " << split_line.size() << std::endl;
129  continue;
130  }
131 
132  // read the start and goal positions
133  try
134  {
135  const int start_x = stoi(split_line[4]);
136  const int start_y = stoi(split_line[5]);
137  const int goal_x = stoi(split_line[6]);
138  const int goal_y = stoi(split_line[7]);
139 
140  // store the positions in the list
141  start_positions.emplace_back(start_x, start_y);
142  start_locations.push_back(map_data.position_to_index(start_positions.back()));
143  goal_positions.emplace_back(goal_x, goal_y);
144  goal_locations.push_back(map_data.position_to_index(goal_positions.back()));
145  loaded_agents++;
146  }
147  catch (const std::exception& e)
148  {
149  throw std::runtime_error("Could not read start and goal positions: '" + (std::string)e.what() + "'");
150  }
151  }
152 
153  // cloase the map file
154  scene_file.close();
155 
156  // check whether enough agents were loaded
157  if (loaded_agents != num_of_agents)
158  {
159  std::cout << "WARNING: Unable to load " << num_of_agents << " agents, as only " << loaded_agents
160  << " agents are available in the scene file." << std::endl;
161  num_of_agents = loaded_agents;
162  }
163 }

174 void Instance::calculate_heuristics(bool calculate_manhattan, bool calculate_euclidean)
175 {
176 
177  // check map validity
178  assertm(map_data.width * map_data.height > 0, "Invalid map.");
179  assertm((int)map_data.data.size() == map_data.width * map_data.height, "Map size invalid");
180  assertm(map_data.get_num_free_cells() > 0 && map_data.get_num_free_cells() <= (int)map_data.data.size(), "Invalid free cell count.");
181 #ifndef CALCULATE_OTHER_HEURISTICS
182  assertm(calculate_euclidean == false || calculate_manhattan == false, "Heuristics turned off during copimle.");
183 #endif
184 
185  const std::vector<int> int_initializer(map_data.width * map_data.height, -1);
186 #ifdef CALCULATE_OTHER_HEURISTICS
187  if (calculate_euclidean)
188  {
189  // clear the vectors
190  heuristic_euclidean.clear();
191  // preallocate vectors for the heuristics
192  const std::vector<double> euclidean_initializer(map_data.width * map_data.height, 0.0);
193  heuristic_euclidean.resize(num_of_agents, euclidean_initializer);
194  }
195 
196  if (calculate_manhattan)
197  {
198  heuristic_manhattan.clear();
199  heuristic_manhattan.resize(num_of_agents, int_initializer);
200  }
201 
202 
203  // fill the vectors
204  if (calculate_euclidean || calculate_manhattan)
205  {
206 #pragma omp parallel for collapse(2)
207  for (int i = 0; i < num_of_agents; i++) // iterate over agents
208  {
209  for (int y = 0; y < map_data.height; y++) // iterate over all positions
210  {
211  const Point2d& goal = goal_positions[i];
212  for (int x = 0; x < map_data.width; x++)
213  {
214  if (calculate_manhattan)
215  {
216  heuristic_manhattan[i][y * map_data.width + x] = abs(goal.x - x) + abs(goal.y - y);
217  }
218  if (calculate_euclidean)
219  {
220  heuristic_euclidean[i][y * map_data.width + x] = std::hypot(goal.x - x, goal.y - y);
221  }
222  }
223  }
224  }
225  }
226 #endif
227 
228  // precompute the distance heuristic
229  heuristic_distance.clear();
230  heuristic_distance.resize(num_of_agents, int_initializer);
231  calculate_distance_heuristic();
232 }

255 void Instance::calculate_distance_heuristic()
256 {
257  // calculate distance from each goal for each position
258 #pragma omp parallel for
259  for (int i = 0; i < (int)goal_locations.size(); i++)
260  {
261  const int goal = goal_locations[i];
262  std::vector<int>& curr = heuristic_distance[i];
263 
264  // by default the first integer is used as priority, and the elements with highest priority are popped first
265  std::priority_queue<std::pair<int, int>> open_list;
266 
267  // create first node which is the goal
268  curr[goal] = 0;
269  open_list.push({0, goal});
270 
271  // find distance from the goal for each position
272  while (!open_list.empty())
273  {
274  // pop the element with the highest priority
275  auto [priority, location] = open_list.top();
276  open_list.pop();
277 
278  // find the neighbor locations
279  const std::vector<int>& curr_neighbors = get_neighbor_locations(location);
280  for (int neighbor : curr_neighbors)
281  {
282  // if the neighbor is not visited yet and is free cell
283  if (curr[neighbor] == -1 && map_data.index(neighbor) == 0)
284  {
285  // calculate the distance
286  curr[neighbor] = curr[location] + 1;
287  open_list.push({-curr[neighbor], neighbor}); // use negative priority to get the elements closest to goal first
288  }
289  }
290  }
291  }
292 }
