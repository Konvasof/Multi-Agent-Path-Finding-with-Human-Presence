30 void ConstraintTable::add_constraint(const TimePoint& timepoint, int agent_num)
31 {
32  // check the location is valid
33  assertm(instance.get_map_data().is_in(timepoint.location), "Invalid location.");
34  assertm(agent_num >= 0 && agent_num < instance.get_num_of_agents(), "Invalid agent number.");
35  int free_location = instance.location_to_free_location(timepoint.location);
36  std::list<std::pair<TimeInterval, int>>& cur_TI_list = constraints[free_location];
37 
38  bool inserted = false;
39  // find where to insert the interval
40  for (auto it = cur_TI_list.cbegin(); it != cur_TI_list.cend(); it++)
41  {
42  // check that the intervals do not have any overlap
43 
44  assertm(!overlap((*it).first, timepoint.interval), "Cannot add overlapping constraints.");
45 
46  // insert before the first interval that starts later
47  if (it->first.t_min > timepoint.interval.t_min)
48  {
49  // check that there is no overlap
50  // assertm(timepoint.interval.t_max < it->t_min, "Cannot add overlapping constraints.");
51  cur_TI_list.insert(it, {timepoint.interval, agent_num});
52  inserted = true;
53  break;
54  }
55  }
56 
57  // if no interval starts later, insert to the back of the list
58  if (!inserted)
59  {
60  cur_TI_list.emplace_back(timepoint.interval, agent_num);
61  }
62 
63  // add to the set of visiting agents
64  // if the agent is not yet in the set, add it with count 1
65  if (agents_counts[free_location].find(agent_num) == agents_counts[free_location].end())
66  {
67  agents_counts[free_location][agent_num] = 1;
68  }
69  else
70  {
71  // else increase the count
72  agents_counts[free_location][agent_num] += 1;
73  }
74 }

76 void ConstraintTable::remove_constraint(const TimePoint& timepoint, int agent_num)
77 {
78  // check the location is valid
79  assertm(instance.get_map_data().is_in(timepoint.location), "Invalid location.");
80  int free_location = instance.location_to_free_location(timepoint.location);
81 
82 
83  std::list<std::pair<TimeInterval, int>>& cur_TI_list = constraints[free_location];
84 
85  // check that the list is not empty
86  assertm(!cur_TI_list.empty(), "Trying to remove interval from an empty list.");
87 
88  assertm(agents_counts[free_location].find(agent_num) != agents_counts[free_location].end(),
89  "Removing agent that is not in the agents counts");
90  agents_counts[free_location][agent_num] -= 1;
91  // if the count drops to 0, erase the agent number from keys
92  if (agents_counts[free_location][agent_num] == 0)
93  {
94  agents_counts[free_location].erase(agent_num);
95  }
96 
97  // remove the interval from the list
98  for (auto it = cur_TI_list.cbegin(); it != cur_TI_list.cend(); it++)
99  {
100  if ((*it).first == timepoint.interval)
101  {
102  cur_TI_list.erase(it);
103  return;
104  }
105  }
106  // if the interval was not found, throw an error
107  throw std::runtime_error("Trying to remove non-existing interval from the constraint table.");
108 }

193 auto ConstraintTable::get_blocking_agent(int from, int to, int time) const -> std::pair<int, int>
194 {
195  // check the location is valid
196  assertm(instance.get_map_data().is_in(from) && instance.get_map_data().is_in(to), "Invalid location.");
197 
198  // check the time is valid
199  assertm(time >= 0, "Invalid time.");
200 
201  int to_free = instance.location_to_free_location(to);
202 
203  const std::list<std::pair<TimeInterval, int>>& cur_TI_list = constraints[to_free];
204 
205  // if there are no constraints, return -1 (no agent blocking the square)
206  if (cur_TI_list.empty())
207  {
208  return std::make_pair(-1, -1);
209  };
210 
211  // find the constraint
212  int vertex_constraint = -1;
213  for (const auto& it : cur_TI_list)
214  {
215  const TimeInterval& constraint_tp = it.first;
216  if (constraint_tp.t_min <= time && constraint_tp.t_max >= time)
217  {
218  // return the agent that created the constraint
219  vertex_constraint = it.second;
220  break;
221  }
222  // skip constraints that are too late
223  if (constraint_tp.t_min > time)
224  {
225  break;
226  }
227  }
228 
229  // find edge constraint
230  int edge_constraint = -1;
231  if (from != to)
232  {
233  // we want to look for opposite edge, because there can not be any edge ending in to location without location constraint
234  edge_constraint = edge_constraint_table.get(to, from, time);
235  }
236  return std::make_pair(vertex_constraint, edge_constraint);
237 }

239 auto ConstraintTable::get_blocking_agents(int location, int time_min) const -> std::vector<int>
240 {
241  // check the location is valid
242  assertm(instance.get_map_data().is_in(location), "Invalid location.");
243 
244  // check the time is valid
245  assertm(time_min >= 0, "Invalid time.");
246 
247  int to_free = instance.location_to_free_location(location);
248 
249  const std::list<std::pair<TimeInterval, int>>& cur_TI_list = constraints[to_free];
250 
251  // if there are no constraints, return -1 (no agent blocking the square)
252  assertm(!cur_TI_list.empty(), "Trying to get blocking agents from an empty list.");
253 
254  std::unordered_set<int> blocking_agents = {};
255  std::vector<int> blocking_agents_vec = {};
256  assertm(cur_TI_list.back().first.t_max == INT_MAX, "Last interval should be infinite.");
257 
258  // check how many intervals are present
259  if (cur_TI_list.size() < 2)
260  {
261  return {};
262  }
263 
264  // iterate over the constraints from end to beginning (skip the last interval, as agent can not block itself)
265  for (auto it = std::next(cur_TI_list.rbegin()); it != cur_TI_list.rend(); it++)
266  {
267  // std::cout << "Interval: " << it->first << ", agent: " << it->second << std::endl;
268  if (it->first.t_max < time_min)
269  {
270  // there can not be any more blocking agents
271  break;
272  }
273 
274  // check whether the agent was not already added
275  const int agent_num = it->second;
276  if (blocking_agents.find(agent_num) == blocking_agents.end())
277  {
278  // add the agent to the set
279  blocking_agents.insert(agent_num);
280  blocking_agents_vec.push_back(agent_num);
281  }
282  }
283  return blocking_agents_vec;
284 }

321 void ConstraintTable::build_parallel_2(const std::vector<TimePointPath>& paths)
322 {
323  // calculate how many constraints to add
324  int total_work = 0;
325  for (const auto& path : paths)
326  {
327  total_work += static_cast<int>(path.size());
328  }
329  assertm(total_work > 0, "No constraints to add.");
330 
331  std::vector<const TimePoint*> timepoints(total_work);
332  std::vector<int> agent_nums(total_work);
333  std::vector<int> prev_location(total_work, -1);
334  int idx = 0;
335  for (int i = 0; i < static_cast<int>(paths.size()); i++)
336  {
337  const auto& path = paths[i];
338  for (int j = 0; j < (int)path.size(); j++)
339  {
340  timepoints[idx] = &path[j];
341  agent_nums[idx] = i;
342  if (j > 0)
343  {
344  prev_location[idx] = path[j - 1].location;
345  }
346  idx++;
347  }
348  }
349 
350 #pragma omp parallel for
351  for (int i = 0; i < total_work; i++)
352  {
353  // add the point constraint
354  omp_set_lock(&locks.locks[instance.location_to_free_location(timepoints[i]->location)]);
355  add_constraint(*timepoints[i], agent_nums[i]);
356  // add the edge constraint
357  if (prev_location[i] != -1) // skip the first point as there is no edge
358  {
359  // the time when the edge is used is the start time of the later interval
360  edge_constraint_table.add(prev_location[i], timepoints[i]->location, timepoints[i]->interval.t_min, agent_nums[i]);
361  }
