210 auto timepointpath_to_path(const TimePointPath& tp_path) -> Path
211 {
212  Path path;
213  for (auto it = tp_path.begin(); it != tp_path.end(); it++)
214  {
215  // for last element add the position only once, as it is obvious that the agent stays in the last position forever
216  int N = 1;
217 
218  // find the start of the next time interval
219  if (it != std::prev(tp_path.end()))
220  {
221  N = std::next(it)->interval.t_min - it->interval.t_min;
222  }
223 
224  // insert N copies to the end of the vector
225  path.resize(path.size() + N, it->location);
226  }
227  return path;
228 }

230 auto path_to_timepointpath(const Path& path) -> TimePointPath
231 {
232  if (path.empty())
233  {
234  return TimePointPath();
235  }
236 
237  TimePointPath tp_path;
238  TimePoint curr(path.front(), {0, -1});
239  for (int it : path)
240  {
241  // if the new location is not the same as the last location, create new interval
242  if (it != curr.location)
243  {
244  tp_path.push_back(curr);
245 
246  // set the new location
247  curr.location = it;
248  curr.interval.t_min = curr.interval.t_max + 1;
249  curr.interval.t_max = curr.interval.t_min;
250  }
251  else
252  {
253  curr.interval.t_max++;
254  }
255  }
256  // the agent stays on the last position
257  curr.interval.t_max = INT_MAX;
258  tp_path.push_back(curr);
259  return tp_path;
260 }

294 auto is_valid_timepointpath(const TimePointPath& tp_path) -> bool
295 {
296  // special case - empty path
297  if (tp_path.empty())
298  {
299  return true;
300  }
301 
302  // check that each interval has smaller t_min than t_max
303  for (const auto& it : tp_path)
304  {
305  if (it.interval.t_min > it.interval.t_max)
306  {
307  return false;
308  }
309  }
310 
311  // check that the agent stays at its last position
312  if (tp_path.back().interval.t_max != INT_MAX)
313  {
314  return false;
315  }
316 
317  if ((int)tp_path.size() >= 2)
318  {
319  // check that each interval starts after the previous one ends
320  for (auto it = tp_path.begin(); it != std::prev(tp_path.end()); it++)
321  {
322  if (it->interval.t_max != std::next(it)->interval.t_min - 1)
323  {
324  return false;
325  }
326  }
327  }

329  // check whether the locations are neighbors
330  return true;
