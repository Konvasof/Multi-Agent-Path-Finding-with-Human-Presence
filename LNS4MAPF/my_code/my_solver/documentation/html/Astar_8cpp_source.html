81 Path ASTAR::find_path(const Point2d& start, const Point2d& goal, const Map& map_data)
82 {
83  Path path;
84 
85  // check whether start and goal inside the map
86  if (!map_data.is_in(start) || !map_data.is_in(goal)) return path;
87 
88  // check whether start and goal not obstructed
89  if (map_data.index(start) == 1 && map_data.index(goal) == 1) return path;
90 
91  // calculate the heuristic
92  double h = std::hypot(goal.x - start.x, goal.y - start.y);
93 
94  // create frontier - possible datastructures priority_queue, ordered set and pairing heap, TODO try all of them
95  ASTAR::PriorityQueue open_list;
96 
97  // create the start node
98  open_list.push(std::make_shared<AstarNode>(start.x, start.y, 0, h, nullptr));
99 
100  // create known nodes database
101  ASTAR::UnorderedSet known;
102 
103  // main loop
104  while (!open_list.empty())
105  {
106  // pop the node to be processed
107  std::shared_ptr<AstarNode> current = std::move(open_list.top());
108  open_list.pop();
109 
110  // check whether node is in known set
111  if (known.find(current->position) != known.end()) continue;
112 
113  // check whether goal is reached
114  if (current->position.x == goal.x && current->position.y == goal.y)
115  {
116  // extract the path
117  return ASTAR::extract_path(current);
118  }
119 
120  // add to known nodes
121  known.insert(current->position);
122 
123  // expand current node
124  std::vector<Point2d> neighbors = map_data.find_neighbors(current->position);
125  for (Point2d& neighbor : neighbors)
126  {
127  // check whether neighbor already known
128  if (known.find(neighbor) != known.end()) continue;
129 
130  double h = std::hypot(goal.x - neighbor.x, goal.y - neighbor.y);
131  // add to the open list
132  open_list.push(std::make_shared<AstarNode>(neighbor, current->g + 1, h, current));
133  }
134  }
135 
136  // could not find a path
137  return {};
138 }
