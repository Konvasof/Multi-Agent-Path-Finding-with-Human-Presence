54 void Experiment::run()
55 {
56  // iterate over all maps
57  for (int i = 0; i < static_cast<int>(static_cast<int>(maps.size())); i++)
58  {
59  // reset the map progress bars
60  if (show_progress)
61  {
62  multi_progress_bar->set_progress<0>(static_cast<size_t>(i));
63  multi_progress_bar->set_option<0>(
64  indicators::option::PostfixText{std::to_string(i) + "/" + std::to_string(static_cast<int>(maps.size())) + " "});
65  }
66 
67  double time_limit = time_limits[i];
68 
69  // skip maps that were already solved
70  const int start_offset_map = sum_agent_nums * runs_per_map * static_cast<int>(algorithms.size());
71  sum_agent_nums += static_cast<int>(agent_nums[i].size());
72  const int end_offset_map = sum_agent_nums * runs_per_map * static_cast<int>(algorithms.size());
73 
74  if (std::all_of(skips.begin() + start_offset_map, skips.begin() + end_offset_map, [](bool b) { return b; }))
75  {
76  if (show_progress)
77  {
78  multi_progress_bar->set_progress<1>(agent_nums[i].size());
79  multi_progress_bar->set_progress<2>(static_cast<size_t>(runs_per_map));
80  }
81  skipped_maps++;
82  skipped_agent_nums += static_cast<int>(agent_nums[i].size());
83  skipped_runs += static_cast<int>(agent_nums[i].size()) * runs_per_map;
84  skipped_implementations += static_cast<int>(agent_nums[i].size()) * runs_per_map * static_cast<int>(algorithms.size());
85  continue;
86  }
87 
88  // retrieve map name and scene names
89  const std::string map_name = maps[i];
90  const std::vector<std::string> scenes = get_scene_names(map_name, runs_per_map);
91 
92  if (show_progress)
93  {
94  multi_progress_bar->set_option<1>(indicators::option::MaxProgress{(int)agent_nums[i].size()});
95  }
96  // iterate over agent numbers
97  for (int j = 0; j < static_cast<int>(agent_nums[i].size()); j++)
98  {
99  // update the progress bar iteration number display
100  if (show_progress)
101  {
102  multi_progress_bar->set_option<1>(
103  indicators::option::PostfixText{std::to_string(j) + "/" + std::to_string(agent_nums[i].size()) + " "});
104  multi_progress_bar->set_progress<1>(static_cast<size_t>(j));
105  }
106 
107  // skip agent_nums that were already solved
108  const int start_offset_an = start_offset_map + j * runs_per_map * static_cast<int>(algorithms.size());
109  const int end_offset_an = start_offset_map + (j + 1) * runs_per_map * static_cast<int>(algorithms.size());
110  if (std::all_of(skips.begin() + start_offset_an, skips.begin() + end_offset_an, [](bool b) { return b; }))
111  {
112  if (show_progress)
113  {
114  multi_progress_bar->set_progress<2>((size_t)runs_per_map);
115  }
116  skipped_agent_nums++;
117  skipped_runs += runs_per_map;
118  skipped_implementations += runs_per_map * static_cast<int>(algorithms.size());
119  continue;
120  }
121 
122  int agent_num = agent_nums[i][j];
123 
124  // load all instances TODO load only instances that will not be skipped
125  auto [instances, preprocessing_times] = load_instances(scenes, map_name, agent_num);
126 
127  // make sure every map gets enough runs
128  for (int k = 0; k < runs_per_map; k++)
129  {
130  // update the progress bar iteration number display
131  if (show_progress)
132  {
133  multi_progress_bar->set_option<2>(
134  indicators::option::PostfixText{std::to_string(k) + "/" + std::to_string(runs_per_map) + " "});
135  multi_progress_bar->set_progress<2>(static_cast<size_t>(k));
136  }
137 
138  // skip runs that were already solved
139  const int start_offset_run = start_offset_an + k * static_cast<int>(algorithms.size());
140  const int end_offset_run = start_offset_an + (k + 1) * static_cast<int>(algorithms.size());
141  if (std::all_of(skips.begin() + start_offset_run, skips.begin() + end_offset_run, [](bool b) { return b; }))
142  {
143  skipped_runs++;
144  skipped_implementations += static_cast<int>(algorithms.size());
145  continue;
146  }
147 
148  // choose instance
149  assertm(scenes.size() == instances.size(), "The size of scenes and instances must be the same.");
150  const int idx = k % static_cast<int>(scenes.size());
151  const std::string& scene_name = scenes[idx];
152  const Instance& instance = instances[idx];
153  // check whether enough agents loaded
154  if (instance.get_num_of_agents() != agent_num)
155  {
156  throw std::runtime_error("Loaded " + std::to_string(instance.get_num_of_agents()) + " agents instead of " +
157  std::to_string(agent_num));
158  }
159 
160  // generate a random seed (same for all implementations) TODO - for experiments found in the database use the same seed
161  int seed = 0;
162  // look into the database for a seed
163  size_t dot_pos = scene_name.find('.');
164  if (dot_pos == std::string::npos)
165  {
166  throw std::runtime_error("Could not find dot in scene name: '" + scene_name + "'");
167  }
168  const std::string scene_name_stripped = scene_name.substr(map_name.size() + 1, scene_name.find('.') - map_name.size() - 1);
169  int seed_from_db = get_seed_from_db(agent_num, map_name, scene_name_stripped, k / static_cast<int>(scenes.size()));
170  // std::cout << "Seed from db: " << seed_from_db << std::endl;
171  if (seed_from_db >= 0)
172  {
173  seed = seed_from_db;
174  }
175  else
176  {
177  seed = dist(rnd_gen);
178  }
179 
180  // load the initial solution
181  Solution init_sol;
182  if (load_init_sol)
183  {
184  // get the file name
185  std::string initsol_path = get_initsol_path(agent_num, map_name, scene_name_stripped, k / static_cast<int>(scenes.size()));
186  // load the solution
187  init_sol.load(initsol_path, instance);
188  if (!init_sol.feasible)
189  {
190  throw std::runtime_error("Loaded infeasile initial solution.");
191  }
192  // calculate cost
193  init_sol.calculate_cost(instance);
194  }
195 
196  // iterate over all implementations
197  for (int l = 0; l < static_cast<int>(algorithms.size()); l++)
198  {
199  // skip already solved
200  if (skips[start_offset_run + l])
201  {
202  skipped_implementations++;
203  continue;
204  }
205 
206 
207  Algorithm& algo = algorithms[l];
208 
209  // create the seeded random generator (for experiment reproducibility)
210  auto rnd_generator = std::mt19937(seed);
211 
212  // prepare the solver
213  LNS solver(instance, rnd_generator, nullptr, algo.lns_settings);
214 
215  // change the time limit to the time limit of this map
216  solver.settings.time_limit = time_limit;
217 
218  // use the loaded initial solution
219  if (load_init_sol)
220  {
221  solver.solution = init_sol;
222  solver.found_initial_solution = true;
223 
224  // build the safe interval table
225  solver.planner->safe_interval_table.build_sequential(init_sol.paths);
226 
227  // add to the log
228  solver.log.bsf_solution_cost.push_back(init_sol.sum_of_costs);
229  solver.log.bsf_makespan.push_back(init_sol.makespan);
230  solver.log.used_operator.push_back(DESTROY_TYPE::ADAPTIVE);
231  solver.log.iteration_time_wall.push_back(0);
232  solver.log.iteration_time_cpu.push_back(0);
233  }
234 
235  // run the experiment function and measure its execution time
236  Clock clock;
237  clock.start();
238  experiment_function(solver);
239  auto [pp_duration_wall, pp_duration_cpu] = clock.end();
240 
241  // calculate the cost
242  solver.solution.calculate_cost(instance);
243 
244  // write the results of the experiment
245  const std::string algorithm_name = algo.get_name();
246  nlohmann::json experiment_res = {{"experiment_name", experiment_name},
247  {"map_name", map_name},
248  {"scene_name", scene_name},
249  {"num_agents", agent_num},
250  {"algo_name", algorithm_name},
251  {"algo_parameters", algo.get_parameters_str()},
252  {"feasible", solver.solution.feasible},
253  {"sum_of_dist", instance.get_sum_of_distances()},
254  {"sum_of_cost", solver.log.bsf_solution_cost},
255  {"makespan", solver.log.bsf_makespan},
256  {"preprocessing_time_wall", preprocessing_times[idx].first},
257  {"preprocessing_time_cpu", preprocessing_times[idx].second},
258  {"experiment_time_wall", pp_duration_wall},
259  {"experiment_time_cpu", pp_duration_cpu},
260  {"iteration_time_wall", solver.log.iteration_time_wall},
261  {"iteration_time_cpu", solver.log.iteration_time_cpu},
262  {"operators", solver.log.get_used_operator_str()},
263  {"expanded", solver.get_num_of_expanded_nodes()},
264  {"generated", solver.get_num_of_generated_nodes()},
265  {"seed", seed}};
266 
267  // save the experiment
268  save_experiment(experiment_res, agent_num, map_name, scene_name, experiment_name, algorithm_name);
269  }
270  }
271  }
272  }
273  // mark all progress bars as finished
274  if (show_progress)
275  {
276  multi_progress_bar->set_option<0>(indicators::option::PostfixText{std::to_string(static_cast<int>(maps.size())) + "/" +
277  std::to_string(static_cast<int>(maps.size())) + " "});
278  multi_progress_bar->set_progress<0>(maps.size());
279  multi_progress_bar->set_progress<1>(agent_nums.back().size());
280  multi_progress_bar->set_option<1>(indicators::option::PostfixText{std::to_string(agent_nums.back().size()) + "/" +
281  std::to_string(agent_nums.back().size()) + " "});
282  multi_progress_bar->set_option<2>(
283  indicators::option::PostfixText{std::to_string(runs_per_map) + "/" + std::to_string(runs_per_map) + " "});
284  multi_progress_bar->set_progress<2>(static_cast<size_t>(runs_per_map));
285  }
286 
287  std::cout << "Experiment finished. Skipped: " << skipped_maps << "/" << static_cast<int>(maps.size()) << " maps, " << skipped_agent_nums
288  << "/" << sum_agent_nums << " agent_nums, " << skipped_runs << "/" << sum_agent_nums * runs_per_map << " runs, "

293 auto Experiment::get_seed_from_db(int agent_num, const std::string& map_name, const std::string& scene_name_stripped, int run) -> int
294 {
295  // Query the database
296  std::string command = "SELECT seed FROM " + experiment_name + " WHERE file_name = ?";
297  // std::string command = "SELECT seed FROM experiments";
298  std::string experiment_filename;
299  if (experiment_name == "sipp_pp")
300  {
301  experiment_filename = experiment_name + "_" + std::to_string(agent_num) + "agents_" + map_name + "-" + scene_name_stripped +
302  "SIPP_lns_orig_" + std::to_string(run);
303  }
304  else if (experiment_name == "influence_of_p" || experiment_name == "influence_of_w" || experiment_name == "repair" ||
305  experiment_name == "influence_of_ap" || experiment_name == "overall_improvement")
306  {
307  return -1; // w and p are only in my solver
308  }
309  else if (experiment_name == "destroy")
310  {
311  experiment_filename = experiment_name + "_" + std::to_string(agent_num) + "agents_" + map_name + "_" + scene_name_stripped +
312  "_destroy_type_ORIGO_Adaptive_neighborhood_size_4_" + std::to_string(run);
313  }
314  else if (experiment_name == "repair")
315  {
316  experiment_filename =
317  experiment_name + "_" + std::to_string(agent_num) + "agents_" + map_name + "_" + scene_name_stripped + "_" + std::to_string(run);
318  }
319  else if (experiment_name == "feasible" || experiment_name == "feasible_restarts")
320  {
321  experiment_filename = experiment_name + "_" + std::to_string(agent_num) + "agents_" + map_name + "-" + scene_name_stripped +
322  "SIPP_lns_orig_" + std::to_string(run);
323  }
324  else
325  {
326  throw std::runtime_error("Unknown experiment name: " + experiment_name);
327  }
328 
329  // std::cout << "Filename: '" << experiment_filename << "'" << std::endl;
330  sqlite3_stmt* stmt = nullptr;
331 
332  int rc = sqlite3_prepare_v2(db, command.c_str(), -1, &stmt, nullptr);
333  if (rc == SQLITE_OK) // fetched data
334  {
335  rc = sqlite3_bind_text(stmt, 1, experiment_filename.c_str(), experiment_filename.size(), SQLITE_TRANSIENT);
336  if (rc != SQLITE_OK)
337  {
338  throw std::runtime_error("Unable to bind text in the sql statement.");
339  }
340 
341  if (sqlite3_step(stmt) == SQLITE_ROW)
342  {
343  int seed = sqlite3_column_int(stmt, 0);
344  if (seed < 0)
345  {
346  throw std::runtime_error("Invalid seed: '" + std::to_string(seed) + "'");
347  }
348  sqlite3_finalize(stmt);
349  return seed;
350  }
351  }
352  else
353  {
354  // the experiment table does not exist in the database yet
355  }
356 
357  sqlite3_finalize(stmt);
358  return -1;
359 }
