516 void Visualizer::update_path_vis(const int agent_num, const Path& path)
517 {
518  assertm(agent_num >= 0 && agent_num < (int)agents_colors.size() && agent_num < instance.get_num_of_agents(), "Invalid agent number.");
519 
520  // check whether the vector is big enough
521  if (agent_num >= (int)agents_paths_vis.size())
522  {
523  agents_paths_vis.resize(agent_num + 1);
524  }
525 
526  // find the color for this agent
527  const sf::Color color = agents_colors[agent_num];
528 
529  // create the path visualization
530  sf::VertexArray path_vis(sf::Quads);
531  if (path.size() >= 2)
532  {
533  auto segment_start = path.cbegin();
534  auto segment_end = path.cbegin() + 1;
535  enum Direction dir = Direction::NONE;
536 
537  // merge contiguous segments
538  for (int i = 2; i < (int)path.size() + 1; i++)
539  {
540  if (segment_start == segment_end || *segment_start == *segment_end) // segment of length 0
541  {
542  segment_end++;
543  continue;
544  }
545  // skip repeating points
546  if (i < (int)path.size() && path[i] == *segment_end)
547  {
548  segment_end++;
549  continue;
550  }
551 
552  if (dir == Direction::NONE)
553  {
554  dir = find_direction(*segment_start, *segment_end);
555  }
556 
557  assertm(dir != Direction::NONE, "Direction must be set.");
558 
559  // check whether segment continuous (trick with checking if i == path.size() is used to prevent out of bounds)
560  if (i != (int)path.size() && dir == find_direction(*segment_end, path[i]))
561  {
562  // extend the end of the segment
563  segment_end++;
564  continue;
565  }
566 
567  // find the rectangle that represents the segment
568  Point2d segment_start_pt = instance.location_to_position(*segment_start);
569  Point2d segment_end_pt = instance.location_to_position(*segment_end);
570 
571  int min_x, min_y, max_x, max_y;
572  min_x = std::min(segment_start_pt.x, segment_end_pt.x);
573  min_y = std::min(segment_start_pt.y, segment_end_pt.y);
574  max_x = std::max(segment_start_pt.x, segment_end_pt.x);
575  max_y = std::max(segment_start_pt.y, segment_end_pt.y);
576  assertm((max_x - min_x > 0) != (max_y - min_y > 0), "Segment must be either horizontal or vertical.");
577 
578  // calculate the corners of the rectangle
579  sf::Vector2f left_upper =
580  sf::Vector2f(CELL_SIZE * (min_x + 0.5 - (double)1 / (2 * LINE_WIDTH)), CELL_SIZE * (min_y + 0.5 - (double)1 / (2 * LINE_WIDTH)));
581  sf::Vector2f right_upper =
582  sf::Vector2f(CELL_SIZE * (max_x + 0.5 + (double)1 / (2 * LINE_WIDTH)), CELL_SIZE * (min_y + 0.5 - (double)1 / (2 * LINE_WIDTH)));
583  sf::Vector2f right_lower =
584  sf::Vector2f(CELL_SIZE * (max_x + 0.5 + (double)1 / (2 * LINE_WIDTH)), CELL_SIZE * (max_y + 0.5 + (double)1 / (2 * LINE_WIDTH)));
585  sf::Vector2f left_lower =
586  sf::Vector2f(CELL_SIZE * (min_x + 0.5 - (double)1 / (2 * LINE_WIDTH)), CELL_SIZE * (max_y + 0.5 + (double)1 / (2 * LINE_WIDTH)));
587 
588  // add the segment into the path visualization
589  path_vis.append(sf::Vertex(left_upper, color));
590  path_vis.append(sf::Vertex(right_upper, color));
591  path_vis.append(sf::Vertex(right_lower, color));
592  path_vis.append(sf::Vertex(left_lower, color));
593 
594  // start a new segment
595  segment_start = segment_end;
596  segment_end++;
597  dir = find_direction(*segment_start, *segment_end);
598  }
599  }
600  assertm(agent_num < (int)agents_paths_vis.size(), "Agent_num must be bigger than the size of the paths visualization vector.");
601  agents_paths_vis[agent_num] = path_vis;
602 }

759 void Visualizer::adjust_sfml_part_viewport(sf::RenderWindow& window, sf::View& view, const float left_panel_size,
760  const float right_panel_size, const float bottom_panel_size)
761 {
762  // calculate the map parameters
763  const int map_width = instance.get_map_data().width;
764  const int map_height = instance.get_map_data().height;
765  const double map_aspect_ratio = (double)map_width / (double)map_height;
766 
767  // get the window parameters
768  const int window_width = window.getSize().x;
769  const int window_height = window.getSize().y;
770 
771  // calculate the size of the part of the window where the sfml stuff will be rendered
772  const double sfml_part_width = window_width * (1.0 - left_panel_size - right_panel_size);
773  const double sfml_part_height = window_height * (1.0 - bottom_panel_size);
774  const double sfml_part_aspect_ratio = sfml_part_width / sfml_part_height;
775  double viewport_width, viewport_height;
776 
777  // the viewport must have the same aspoect ratio as the map
778  if (map_aspect_ratio < sfml_part_aspect_ratio)
779  {
780  // map is wider
781  viewport_width = sfml_part_width;
782  viewport_height = viewport_width / map_aspect_ratio;
783  }
784  else
785  {
786  // map is taller
787  viewport_height = sfml_part_height;
788  viewport_width = viewport_height * map_aspect_ratio;
789  }
790 
791  // calculate the viewport size relative to the window size
792  const auto viewport_width_rel = (float)(viewport_width / sfml_part_width);
793  const auto viewport_height_rel = (float)(viewport_height / sfml_part_height);
794  const float offset_x = (1 - viewport_width_rel) / 2;
795  const float offset_y = (1 - viewport_height_rel) / 2;
796  view.setViewport(sf::FloatRect({left_panel_size + offset_x, offset_y},
797  {(float)(viewport_width / window_width), (float)(viewport_height / window_height)}));

800 void Visualizer::process_events(sf::RenderWindow& window, sf::View& view)
801 {
802  // poll all events from the window
803  sf::Event event{};
804  while (window.pollEvent(event))
805  {
806  ImGui::SFML::ProcessEvent(window, event); // Send event to ImGui
807 
808  // close event
809  if (event.type == sf::Event::Closed)
810  {
811  running = false;
812  }
813  // zoom event
814  else if (event.type == sf::Event::MouseWheelScrolled)
815  {
816  if (event.mouseWheelScroll.delta > 0)
817  {
818  view.zoom(1 / zoom_step); // Zoom in
819  zoom_level *= 1 / zoom_step;
820  }
821  else
822  {
823  view.zoom(zoom_step); // Zoom out
824  zoom_level *= zoom_step;
825  }
826  }
827  // mouse click
828  else if (event.type == sf::Event::MouseButtonPressed)
829  {
830  if (event.mouseButton.button == sf::Mouse::Left)
831  {
832  const int window_width = window.getSize().x;
833  const int window_height = window.getSize().y;
834  const sf::Vector2i mouse_pos = sf::Vector2i(event.mouseButton.x, event.mouseButton.y);
835  const sf::Vector2f mouse_pos_rel =
836  sf::Vector2f((float)mouse_pos.x / (float)window_width, (float)mouse_pos.y / (float)window_height);
837  // check whether clicked on the map
838  if (view.getViewport().contains(mouse_pos_rel))
839  {
840  const sf::Vector2f translated_pos = window.mapPixelToCoords(mouse_pos);
841  is_moving = true;
842  last_click_pos = mouse_pos;
843  for (int i = 0; i < (int)agents_circles.size(); i++)
844  {
845  // check whether clicked on agent
846  if (agents_circles[i].getGlobalBounds().contains(translated_pos))
847  {
848  selected_agent = i;
849  update_sipp_info();
850  return;
851  }
852  }
853  }
854  }
855  }
856  // mouse is pressed
857  else if (sf::Mouse::isButtonPressed(sf::Mouse::Left))
858  {
859  if (is_moving)
860  {
861  const sf::Vector2i mouse_pos = sf::Mouse::getPosition(window);
862  const sf::Vector2i delta = last_click_pos - mouse_pos;
863  // get viewport size
864  const float viewport_width = view.getViewport().width;
865  const float viewport_height = view.getViewport().height;
866  const float view_width = view.getSize().x;
867  const float view_height = view.getSize().y;
868  const int window_width = window.getSize().x;
869  const int window_height = window.getSize().y;
870  view.move(view_width * delta.x / ((float)window_width * viewport_width),
871  view_height * delta.y / ((float)window_height * viewport_height));
872  last_click_pos = mouse_pos;
873  }
874  }
875  else if (event.type == sf::Event::MouseButtonReleased)
876  {
877  is_moving = false;
878  }
879  // Panning with arrow keys
880  else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Left))
881  {
882  view.move(-CELL_SIZE * zoom_level, 0);
883  }
884  else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Right))
885  {
886  view.move(CELL_SIZE * zoom_level, 0);
887  }
888  else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Up))
889  {
890  view.move(0, -CELL_SIZE * zoom_level);
891  }
892  else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Down))
893  {
894  view.move(0, CELL_SIZE * zoom_level);
895  }
896  }
