17 {
18  // check that each agent has a path
19  if (instance.get_num_of_agents() != (int)paths.size())
20  {
21  // std::cout << "Agent without a path." << std::endl;
22  return false;
23  }
24 
25  convert_paths();
26 
27  for (int i = 0; i < (int)paths.size(); i++)
28  {
29  // check validity of each timepointpath
30  if (!instance.check_timepointpath_validity(paths[i]))
31  {
32  return false;
33  }
34 
35  // convert the path
36  converted_paths[i] = timepointpath_to_path(paths[i]);
37  }
38 
39  // check that each path starts and ends in the correct position and find the longest path
40  int T_max = 0;
41  for (int i = 0; i < (int)converted_paths.size(); i++)
42  {
43  // check that no path has length 0
44  if (converted_paths[i].empty())
45  {
46  // std::cout << "Empty path." << std::endl;
47  return false;
48  }
49 
50  // check that the path starts in the start position
51  if (!(converted_paths[i].front() == instance.get_start_locations()[i]))
52  {
53  // std::cout << "Path does not start in the start position." << std::endl;
54  return false;
55  }
56 
57  // check that the path ends in the goal position
58  if (!(converted_paths[i].back() == instance.get_goal_locations()[i]))
59  {
60  // std::cout << "Path does not end in the goal position." << std::endl;
61  return false;
62  }
63 
64  // remember the length of the longest path
65  if ((int)converted_paths[i].size() > T_max)
66  {
67  T_max = converted_paths[i].size();
68  }
69  }
70  // std::cout << "Max length of path: " << T_max << std::endl;
71 
72  // now check that there are no collisions or invalid points or invalid moves
73  for (int t = 0; t < T_max; t++)
74  {
75  // create reservation set for each time step to check for dynamic coliisions
76  std::unordered_set<int, std::hash<int>> occupied;
77 
78  for (const auto& path : converted_paths)
79  {
80  int idx = t;
81  const int cur_path_len = (int)path.size();
82 
83  // paths that are already finished at time t
84  if (t >= cur_path_len)
85  {
86  // stay at the last point
87  idx = cur_path_len - 1;
88  }
89 
90  if (idx > 0)
91  {
92  // check if the move is valid
93  Point2d p1 = instance.location_to_position(path[idx - 1]);
94  Point2d p2 = instance.location_to_position(path[idx]);
95  if (abs(p2.x - p1.x) + abs(p2.y - p1.y) > 1)
96  {
97  // std::cout << "Invalid move." << std::endl;
98  return false;
99  }
100  }
101 
102  // check the path does not leave the map
103  if (!instance.get_map_data().is_in(path[idx]))
104  {
105  // std::cout << "Path leaves the map in point " << path[idx] << std::endl;
106  return false;
107  }
108 
109  // check collisions with static obstacles
110  if (instance.get_map_data().index(path[idx]) != 0)
111  {
112  // std::cout << "Collision with static obstacle." << std::endl;
113  return false;
114  }
115 
116  // check collisions with dynamic obstacles
117  if (occupied.find(path[idx]) != occupied.end())
118  {
119  // std::cout << "Collision with dynamic obstacle." << std::endl;
120  return false;
121  }
122 
123  // add the point to the reservation set
124  occupied.insert(path[idx]);
125  }
126  }
127 
128  // check that the agents do not swap edges
129  for (int i = 0; i < (int)converted_paths.size(); i++)
130  {
131  for (int j = i + 1; j < (int)converted_paths.size(); j++)
132  {
133  for (int t = 1; t < T_max; t++)
134  {
135  const int cur_path_len_i = (int)converted_paths[i].size();
136  const int cur_path_len_j = (int)converted_paths[j].size();
137 
138  // paths that are already finished at time t
139  if (t >= cur_path_len_i || t >= cur_path_len_j)
140  {
141  continue;
142  }
143 
144  if (converted_paths[i][t] == converted_paths[j][t - 1] && converted_paths[i][t - 1] == converted_paths[j][t])
145  {
146  // std::cout << "Agents swap edges." << std::endl;
147  return false;
148  }
149  }
150  }
151  }
152 
153  return true;
154 }

156 void Solution::load(const std::string& filename, const Instance& instance)
157 {
158  // open the solution file
159  std::ifstream sol_file(filename.c_str());
160 
161  // check if the file is open
162  if (!sol_file.is_open())
163  {
164  throw std::runtime_error("Cannot open file '" + filename + "'");
165  }
166 
167  // read agents one by one
168  std::string line;
169  int expected_agent_num = 0;
170 
171  while (getline(sol_file, line))
172  {
173  // remove leading and trailing spaces
174  line = boost::algorithm::trim_copy(line);
175 
176  // find the agent number
177  const size_t idx_an = line.find("Agent ");
178  if (idx_an == std::string::npos)
179  {
180  throw std::runtime_error("Cannot find agent number in line '" + line + "'");
181  }
182 
183  const size_t idx2_an = line.find(':');
184  if (idx_an == std::string::npos)
185  {
186  throw std::runtime_error("Cannot find : in line '" + line + "'");
187  }
188 
189  // get the agent number
190  int agent_num = 0;
191  try
192  {
193  agent_num = std::stoi(line.substr(idx_an + 6, idx2_an - idx_an - 6));
194  }
195  catch (const std::exception& e)
196  {
197  throw std::runtime_error("Cannot convert agent number in line '" + line + "'");
198  }
199 
200  // check that the agent number is correct
201  if (agent_num != expected_agent_num)
202  {
203  throw std::runtime_error("Unexpected agent number in line '" + line + "'");
204  }
205  expected_agent_num++;
206 
207  // split the line
208  std::vector<std::string> split_line;
209  boost::algorithm::split_regex(split_line, line.substr(idx2_an + 1), boost::regex("->"));
210 
211  // read the path
212  Path path;
213  for (const std::string& point_str : split_line)
214  {
215  if (point_str.empty())
216  {
217  continue;
218  }
219  // remove the leading and trailing spaces
220  std::string point_str_trimmed = boost::algorithm::trim_copy(point_str);
221 
222  // find the opening bracket of the point
223  size_t idx = point_str_trimmed.find('(');
224  if (idx == std::string::npos)
225  {
226  throw std::runtime_error("Cannot find '(' in point '" + point_str + "'");
227  }
228 
229  // find the closing bracket of the point
230  size_t idx2 = point_str_trimmed.find(')');
231  if (idx2 == std::string::npos)
232  {
233  throw std::runtime_error("Cannot find ')' in point '" + point_str + "'");
234  }
235 
236  // find ,
237  size_t idx_comma = point_str_trimmed.find(',');
238  if (idx_comma == std::string::npos)
239  {
240  throw std::runtime_error("Cannot find ',' in point '" + point_str + "'");
241  }
242 
243  // get the coordinates
244  int x, y;
245  try
246  {
247  // load in format (y,x)
248  y = std::stoi(point_str_trimmed.substr(idx + 1, idx_comma - 1));
249  x = std::stoi(point_str_trimmed.substr(idx_comma + 1, idx2 - idx_comma - 1));
250  }
251  catch (const std::exception& e)
252  {
253  throw std::runtime_error("Cannot convert coordinates in line '" + point_str_trimmed +
254  "', x: " + point_str_trimmed.substr(idx + 1, idx_comma - 1) +
255  ", y: " + point_str_trimmed.substr(idx_comma + 1, idx2 - idx_comma - 1));
256  }
257 
258  // add the point to the path
259  path.push_back(instance.position_to_location(Point2d(x, y)));
260  }
261  // add the path to the solution
262  paths.push_back(path_to_timepointpath(path));
263  }
264 
265  // close the file
266  sol_file.close();
267 
268  // check whether the solution is feasible
269  feasible = is_valid(instance);
270 }

299 void Solution::calculate_cost(const Instance& instance)
300 {
301  if (!feasible)
302  {
303  delays.resize(paths.size(), -1);
304  sum_of_delays = -1;
305  sum_of_costs = -1;
306  makespan = -1;
307  return;
308  }
309  assertm(feasible, "Can not calculate cost of infeasible solution");
310  assertm(!paths.empty(), "Can not calculate cost of an empty solution.");
311  delays.resize(paths.size(), 0);
312  sum_of_delays = 0;
313  sum_of_costs = 0;
314  makespan = paths[0].back().interval.t_min;
315 
316  for (int i = 0; i < static_cast<int>(paths.size()); i++)
317  {
318  makespan = std::max(makespan, paths[i].back().interval.t_min);
319  sum_of_costs += paths[i].back().interval.t_min;
320  const int delay = paths[i].back().interval.t_min - instance.get_heuristic_distance(i, instance.get_start_locations()[i]);
321  sum_of_delays += delay;
322  assertm(delay >= 0, "Delay can not be negative.");
323  assertm(sum_of_costs >= delay && sum_of_costs >= makespan, "Sum of costs should be greater than delay and makespan.");
324  delays[i] = delay;
325  }
326 }
