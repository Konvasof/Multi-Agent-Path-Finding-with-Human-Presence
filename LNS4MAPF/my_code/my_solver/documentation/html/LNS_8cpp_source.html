82 void LNS::solve()
83 {
84  // start measuring time
85  Clock clock;
86  clock.start();
87 
88  // calculate initial solution
89  while (!found_initial_solution && clock.get_current_time().first < settings.time_limit)
90  {
91  found_initial_solution = find_initial_solution();
92 
93  // reset the safe interval table and already planned if needed
94  if (!found_initial_solution)
95  {
96  planner->reset();
97  already_planned.clear();
98  }
99 
100  // restart only if the settings allow it
101  if (!settings.restarts)
102  {
103  break;
104  }
105  }
106 
107  if (!found_initial_solution || settings.max_iter == 0)
108  {
109  return;
110  }
111  assertm(found_initial_solution, "Could not find the initial solution");
112  assertm(solution.is_valid(instance), "Found invalid solution");
113 
114  // std::cout << "Initial solution found in " << initial_solution_time << " second." << std::endl;
115 
116  // initialize the constraint table (needed for randomwalk and intersection destroy operator)
117  if (settings.destroy_settings.type != DESTROY_TYPE::RANDOM)
118  {
119  initialize_constraint_table(solution.paths);
120  }
121 
122  // improve the solution
123  while (iteration_num < settings.max_iter && clock.get_current_time().first < settings.time_limit)
124  {
125  // check for visualization thread end
126  if (settings.sipp_settings.info_type == INFO_type::visualisation)
127  {
128  if (shared_data->is_end.load(std::memory_order_acquire))
129  {
130  break;
131  }
132  }
133 
134  iteration_num++;
135 
136  // time the iteration
137  Clock iteration_clock;
138  iteration_clock.start();
139 
140  // copy solution
141  Solution perturbed_sol = solution;
142 
143  // destroy operator
144  destroy_operator.apply(perturbed_sol);
145 
146  // repair operator
147  repair_operator.apply(perturbed_sol);
148 
149  // accept only feasible solutions
150  bool accepted = false;
151  int improvement = 0;
152  if (!perturbed_sol.feasible)
153  {
154  discard_solution(perturbed_sol, solution);
155  // std::cout << "Discarded infeasible solution." << std::endl;
156  }
157  else
158  {
159  // accept only improving solutions
160  perturbed_sol.calculate_cost(instance);
161  improvement = solution.sum_of_delays - perturbed_sol.sum_of_delays;
162 
163  // find out which destroy strategy was used
164  auto destroy_strategy_index_help = magic_enum::enum_index<DESTROY_TYPE>(last_destroy_strategy);
165  int destroy_strategy_index = -1;
166  if (destroy_strategy_index_help.has_value())
167  {
168  destroy_strategy_index = destroy_strategy_index_help.value();
169  }
170  assertm(destroy_strategy_index >= 0 && destroy_strategy_index < static_cast<int>(magic_enum::enum_count<DESTROY_TYPE>()),
171  "Wrong destroy strategy index.");
172 
173  if (improvement <= 0)
174  {
175  // update the weights
176  if (settings.destroy_settings.type == DESTROY_TYPE::ADAPTIVE)
177  {
178  destroy_weights[destroy_strategy_index] = (1 - BLOCKED_REACTION_FACTOR) * destroy_weights[destroy_strategy_index];
179  }
180  if (last_destroy_strategy == DESTROY_TYPE::BLOCKED)
181  {
182  threshold_blocked = std::max((1 - BLOCKED_REACTION_FACTOR) * threshold_blocked, MIN_BLOCKED_THRESHOLD);
183  }
184  discard_solution(perturbed_sol, solution);
185  // std::cout << "Discarded worse solution, difference: " << perturbed_sol.sum_of_costs - solution.sum_of_costs << std::endl;
186  }
187  else
188  {
189  // update the weights
190  if (settings.destroy_settings.type == DESTROY_TYPE::ADAPTIVE)
191  {
192  destroy_weights[destroy_strategy_index] =
193  reaction_factor * static_cast<double>(improvement) / static_cast<double>(settings.destroy_settings.size) +
194  (1 - reaction_factor) * destroy_weights[destroy_strategy_index];
195  }
196  else if (last_destroy_strategy == DESTROY_TYPE::BLOCKED)
197  {
198  threshold_blocked = std::min((1 + reaction_factor) * threshold_blocked, 1.0);
199  }
200  accepted = true;
201  // remember the perturbed solution
202  // std::cout << "Solution improved by: " << solution.sum_of_delays - perturbed_sol.sum_of_delays << std::endl;
203  solution = std::move(perturbed_sol);
204  }
205  }
206  auto [iteration_time_wall, iteration_time_cpu] = iteration_clock.end();
207 
208  // construct LNS iteration info and add to the shared data
209  if (settings.sipp_settings.info_type == INFO_type::visualisation)
210  {
211  // get the pointer to the original perturbed solution (which might have been moved)
212  Solution* sol = &solution;
213  if (!accepted)
214  {
215  sol = &perturbed_sol;
216  }
217  LNSIterationInfo lns_info(iteration_num, accepted, improvement, sipp_info, *sol,
218  (std::string)magic_enum::enum_name<DESTROY_TYPE>(last_destroy_strategy));
219  sipp_info.clear();
220  shared_data->update_lns_info(lns_info);
221  }
222  // log the iteration
223  else if (settings.sipp_settings.info_type == INFO_type::experiment)
224  {
225  log.bsf_solution_cost.push_back(solution.sum_of_costs);
226  log.bsf_makespan.push_back(solution.makespan);
227  log.used_operator.push_back(last_destroy_strategy);
228  log.iteration_time_wall.push_back(iteration_time_wall);
229  log.iteration_time_cpu.push_back(iteration_time_cpu);
230  }
231  }

265 auto LNS::PrioritizedPlanning() -> Solution
266 {
267  Solution sol;
268  sol.paths.resize(instance.get_num_of_agents(), TimePointPath());
269  const int agent_num = instance.get_num_of_agents();
270 
271  // create a sequence from 0 to agent_num
272  std::vector<int> priorities(agent_num);
273  std::iota(priorities.begin(), priorities.end(), 0);
274 
275  // shuffle the sequence to get random priorities
276  std::shuffle(priorities.begin(), priorities.end(), rnd_generator);
277 
278  // initialize iter info
279  if (settings.sipp_settings.info_type == INFO_type::visualisation)
280  {
281  sipp_info.clear();
282  sipp_info.resize(agent_num);
283  }
284 
285  // plan path for each agent according to its priority
286  assertm(already_planned.empty(), "Running Prioritized Planning when some agents were already planned.");
287  for (int i = 0; i < agent_num; i++)
288  {
289  // get agent with priority i
290  int agent_id = priorities[i];
291 
292  // plan path for agent
293  TimePointPath tp_path = planner->plan(agent_id, already_planned);
294 
295  // get the visualization info
296  if (settings.sipp_settings.info_type == INFO_type::visualisation)
297  {
298  sipp_info[agent_id] = std::move(planner->iter_info);
299  planner->iter_info.clear();
300  }
301 
302  // check validity
303  if (tp_path.empty())
304  {
305  // std::cout << "Unable to find initial solution by prioritized planning. " << std::endl;
306  sol.feasible = false;
307  return sol;
308  }
309 
310  // check validity
311  assertm(instance.check_timepointpath_validity(tp_path), "SIPP planned an invalid timepointpath.");
312 
313  // update the safe interval table
314  planner->safe_interval_table.add_constraints(tp_path);
315 
316  // add to the solution
317  sol.paths[agent_id] = tp_path;
318 
319  // add to the already planned agents
320  already_planned.insert(agent_id);
321  }
322  // add priorities to the visualization info
323  if (settings.sipp_settings.info_type == INFO_type::visualisation)
324  {
325  sol.priorities = std::move(priorities);
326  }
327  return sol;
328 }

346 void LNS::destroy_randomwalk(Solution& sol) const
347 {
348  last_destroy_strategy = DESTROY_TYPE::RANDOMWALK;
349 
350  assertm(constraint_table_initialized, "Constraint table is not initialized.");
351  assertm(settings.destroy_settings.size >= 0 && settings.destroy_settings.size <= instance.get_num_of_agents(),
352  "Invalid neighborhood size.");
353  // if neighborhood size the same as number of agents or more, return all agents
354  if (settings.destroy_settings.size >= instance.get_num_of_agents())
355  {
356  destroy_random(sol);
357  return;
358  }
359 
360  // the tabu list
361  static std::unordered_set<int> tabu_list;
362 
363  // find the most delayed agent that is not on the tabu list
364  assertm(static_cast<int>(sol.delays.size()) == instance.get_num_of_agents(),
365  "The length of delay list must be the same as the numebr of agents.");
366 
367  int max_delay = 0;
368  int most_delayed = -1;
369  for (int i = 0; i < instance.get_num_of_agents(); i++)
370  {
371  // skip agents in tabu list
372  if (tabu_list.find(i) != tabu_list.end())
373  {
374  continue;
375  }
376 
377  if (sol.delays[i] > max_delay)
378  {
379  max_delay = sol.delays[i];
380  most_delayed = i;
381  }
382  }
383 
384  // check, whether to reset tabu list
385  if (max_delay == 0)
386  {
387  // reset the tabu list and try again
388  if (!tabu_list.empty())
389  {
390  tabu_list.clear();
391  destroy_randomwalk(sol);
392  return;
393  }
394 
395  // select random agents
396  std::cout << "Using random destroy" << std::endl;
397  destroy_random(sol);
398  return;
399  }
400 
401  assertm(max_delay > 0, "Max delay is invalid.");
402  assertm(most_delayed >= 0 && most_delayed < instance.get_num_of_agents() && sol.delays[most_delayed] == max_delay,
403  "Invalid id of most delayed agent");
404 
405  // update the tabu list
406  tabu_list.insert(most_delayed);
407  if (static_cast<int>(tabu_list.size()) == instance.get_num_of_agents())
408  {
409  tabu_list.clear();
410  }
411 
412  std::set<int> chosen = {
413  most_delayed}; // TODO in the original code they use set, but I think it makes more sense to use either unordered set or vector
414  int upperbound = sol.paths[most_delayed].back().interval.t_min;
415  assertm(upperbound > 0, "Upperbound is too small.");
416  // perform the first randomwalk from start - not in pseudocode, but in MAPF-LNS2 implementation
417  int chosen_agent = most_delayed;
418  if (!random_walk(chosen_agent, settings.destroy_settings.size, sol.paths[chosen_agent][0].location, 0, upperbound, chosen))
419  {
420  // perform the random walk again, maximum 10 iterations
421  for (int i = 0; i < 10; i++)
422  {
423  // choose a random time from the agents path
424  std::uniform_int_distribution<int> dist_time(0, upperbound);
425  int chosen_t = dist_time(rnd_generator);
426 
427  // choose location
428  int rw_start_location = location_at_time(sol.paths[chosen_agent], chosen_t);
429  assertm(rw_start_location >= 0, "Invalid start location of randomwalk");
430 
431  // perform randomwalk
432  if (random_walk(chosen_agent, settings.destroy_settings.size, rw_start_location, chosen_t, upperbound, chosen))
433  {
434  break;
435  }
436  assertm(static_cast<int>(chosen.size()) < settings.destroy_settings.size, "Enough agents were selected, but false returned.");
437 
438  // choose a random agent
439  std::uniform_int_distribution<int> dist(0, chosen.size() - 1);
440  int idx = dist(rnd_generator);
441  assertm(idx >= 0 && idx < static_cast<int>(chosen.size()), "Index out of bounds.");
442  // iterator to the first element
443  auto it = chosen.begin();
444  std::advance(it, idx);
445  chosen_agent = *it;
446  upperbound = sol.paths[chosen_agent].back().interval.t_min;
447  }
448  }
449 
450  // if not enough agents selected, repeat
451  if (static_cast<int>(chosen.size()) <= 1)
452  {
453  destroy_randomwalk(sol);
454  return;
455  }
456  assertm(static_cast<int>(chosen.size()) > 1, "Not enough agents selected.");
457 
458  // but it is implemented like this in MAPF-LNS2
459  // sol.destroyed_paths = std::vector(std::make_move_iterator(chosen.begin()), std::make_move_iterator(chosen.end()));
460  sol.destroyed_paths.assign(chosen.begin(), chosen.end());
461 
462  std::shuffle(sol.destroyed_paths.begin(), sol.destroyed_paths.end(), rnd_generator);
463  assertm(sol.destroyed_paths.size() > 1, "Not enough paths were destroyed.");
464  sol.feasible = false;
465 }

468 auto LNS::random_walk(int agent_num, int neighborhood_size, int start_location, int start_time, int upperbound, std::set<int>& chosen) const
469  -> bool
470 {
471  assertm(static_cast<int>(chosen.size()) < neighborhood_size, "Randomwalk not needed.");
472  int curr = start_location;
473  for (int t = start_time; t < upperbound; t++)
474  {
475  std::vector<int> next_locations = instance.get_neighbor_locations(curr);
476  next_locations.push_back(curr); // agent can also stay
477  std::shuffle(next_locations.begin(), next_locations.end(), rnd_generator);
478  // try all possible neighbors in random order
479  bool moved = false;
480  for (auto loc : next_locations)
481  {
482  int next_h_val = instance.get_heuristic_distance(agent_num, loc);
483  // if lowerbound on the time to reach goal smaller than limit, accept the move
484  if (t + 1 + next_h_val < upperbound)
485  {
486  // get the agent that is in conflict and add it to the neighborhood
487  auto [conflicting_agent_vertex, conflicting_agent_edge] = constraint_table.get_blocking_agent(curr, loc, t + 1);
488 
489  // if there is an agent in collision, add it to the neighborhood
490  if (conflicting_agent_vertex != -1)
491  {
492  chosen.insert(conflicting_agent_vertex);
493  }
494 
495  if (conflicting_agent_edge != -1)
496  {
497  chosen.insert(conflicting_agent_edge);
498  }
499 
500  curr = loc;
501  moved = true;
502  break;
503  }
504  }
505  // test whether enough agents selected
506  if (static_cast<int>(chosen.size()) >= neighborhood_size)
507  {
508  return true;
509  }
510 
511  // if none of the next locations was valid, return
512  if (!moved)
513  {
514  break;
515  }
516  }
517  assertm(static_cast<int>(chosen.size()) <= neighborhood_size, "Enough agents selected, but false returned.");
518  return false;
519 }

521 void LNS::destroy_intersection(Solution& sol) const
522 {
523  last_destroy_strategy = DESTROY_TYPE::INTERSECTION;
524  assertm(constraint_table_initialized, "Constraint table is not initialized.");
525 
526  // find all intersection vertices
527  std::vector<int> intersection_vertices_free;
528  intersection_vertices_free.reserve(instance.get_num_free_cells()); // upperbound on number of intersection vertices
529 
530  for (int i = 0; i < instance.get_num_free_cells(); i++)
531  {
532  const auto& agents_counts = constraint_table.get_agents_counts_free(i);
533  if (static_cast<int>(agents_counts.size()) >= 3)
534  {
535  intersection_vertices_free.push_back(i);
536  }
537  }
538 
539  // create queue
540  std::queue<int> openlist;
541 
542  // pick a random intersection vertex
543  {
544  std::uniform_int_distribution<int> dist(0, intersection_vertices_free.size() - 1);
545  int idx = dist(rnd_generator);
546  int chosen = instance.free_location_to_location(intersection_vertices_free[idx]);
547  openlist.push(chosen);
548  }
549 
550 
551  std::unordered_set<int> neighborhood; // TODO try set
552  std::vector<bool> known(instance.get_num_cells(), false);
553 
554  while (!openlist.empty())
555  {
556  // retrieve current node
557  int current = openlist.front();
558  openlist.pop();
559  // std::cout << "Current: " << current << std::endl;
560  if (known[current])
561  {
562  continue;
563  }
564  known[current] = true;
565 
566  // check whether intersection vertex
567  auto agents_counts = constraint_table.get_agents_counts(current);
568  if (static_cast<int>(agents_counts.size()) >= 3)
569  {
570  // add agents that visit this vertex to the neighborhood
571  get_intersection_agents(neighborhood, current);
572  }
573 
574  // check whether enough agents selected
575  // std::cout << "Selected: " << static_cast<int>(neighborhood.size()) << std::endl;
576  if (settings.destroy_settings.size == static_cast<int>(neighborhood.size()))
577  {
578  break;
579  }
580 
581  // add neighbors to the queue
582  auto neighbors = instance.get_neighbor_locations(current);
583  for (auto it : neighbors)
584  {
585  openlist.push(it);
586  }
587  }
588  assertm(static_cast<int>(neighborhood.size()) > 1, "Not enough agents selected for destroy.");
589 
590  // convert to vector and shuffle
591  sol.destroyed_paths = std::vector(std::make_move_iterator(neighborhood.begin()), std::make_move_iterator(neighborhood.end()));
592  // shuffle the paths
593  std::shuffle(sol.destroyed_paths.begin(), sol.destroyed_paths.end(), rnd_generator);
594  assertm(sol.destroyed_paths.size() > 0, "No paths were destroyed.");
595  sol.feasible = false;
596 }

598 void LNS::get_intersection_agents(std::unordered_set<int>& neighborhood, int current) const
599 {
600  // find maximal T
601  int t_max = constraint_table.get_last_constraint_start(current);
602 
603  // choose random time
604  std::uniform_int_distribution<int> dist(0, t_max);
605  int t = dist(rnd_generator);
606  int delta = 0;
607  while (static_cast<int>(neighborhood.size()) < settings.destroy_settings.size && t + delta <= t_max && t - delta >= 0)
608  {
609  // find the agent, that blocks the cell at time t + delta
610  auto [blocking_agent_1, blocking_agent_2] = constraint_table.get_blocking_agent(current, current, t + delta);
611  // find the agent, that blocks the cell at time t - delta
612  auto [blocking_agent_3, blocking_agent_4] = constraint_table.get_blocking_agent(current, current, t - delta);
613 
614  // add the agents to the neighborhood
615  std::array<int, 4> blocking_agents = {blocking_agent_1, blocking_agent_2, blocking_agent_3, blocking_agent_4};
616  for (auto blocking_agent : blocking_agents)
617  {
618  if (blocking_agent >= 0 && static_cast<int>(neighborhood.size()) < settings.destroy_settings.size)
619  {
620  neighborhood.insert(blocking_agent);
621  }
622  }
623  delta++;
624  }
625 }

627 void LNS::destroy_adaptive(Solution& sol) const
628 {
629  // choose strategy by roulette wheel selection
630  // calculate the sum of weights
631  auto sum_of_weights = std::reduce(destroy_weights.begin(), destroy_weights.end());
632  assertm(sum_of_weights > 0, "Invalid weights.");
633 
634 
635  // choose a random number in the range [0, sum_of_weights]
636  std::uniform_real_distribution<double> dist(0.0, sum_of_weights);
637  double rand_value = dist(rnd_generator);
638 
639  // spin the roulette
640  double cummulative_sum = 0.0;
641  DESTROY_TYPE chosen_type = DESTROY_TYPE::ADAPTIVE;
642  for (int i = 0; i < static_cast<int>(destroy_weights.size()); i++)
643  {
644  cummulative_sum += destroy_weights[i];
645  if (rand_value <= cummulative_sum)
646  {
647  auto chosen = magic_enum::enum_cast<DESTROY_TYPE>(i);
648  if (chosen.has_value())
649  {
650  chosen_type = chosen.value();
651  break;
652  }
653  throw std::runtime_error(std::string("Unknown DESTROY_TYPE ") + std::string(magic_enum::enum_name<DESTROY_TYPE>(chosen_type)) +
654  std::string(" encountered in adaptive destroy."));
655  }
656  }
657  assertm(chosen_type != DESTROY_TYPE::ADAPTIVE && chosen_type != DESTROY_TYPE::RANDOM_CHOOSE, "Did not select any destroy strategy.");
658 
659  // if sum of weights too small, increase it
660  if (sum_of_weights < 0.1)
661  {
662  for (double& destroy_weight : destroy_weights)
663  {
664  destroy_weight *= 10;
665  }
666  }
667  // std::cout << "Weights are: " << std::endl;
668  // for (auto it : destroy_weights)
669  // {
670  // std::cout << it << ", ";
671  // }
672  // std::cout << std::endl;
673  // std::cout << "Chosen destroy: " << magic_enum::enum_name<DESTROY_TYPE>(chosen_type) << std::endl;
674 
675  // apply the destroy operator
676  switch (chosen_type)
677  {
678  case DESTROY_TYPE::RANDOM:
679  destroy_random(sol);
680  break;
681  case DESTROY_TYPE::RANDOMWALK:
682  destroy_randomwalk(sol);
683  break;
684  case DESTROY_TYPE::INTERSECTION:
685  destroy_intersection(sol);
686  break;
687  default:
688  throw std::runtime_error(std::string("Unknown DESTROY_TYPE ") + std::string(magic_enum::enum_name<DESTROY_TYPE>(chosen_type)) +
689  std::string(" encountered in adaptive destroy."));
690  }

693 void LNS::destroy_blocked(Solution& sol) const
694 {
695  last_destroy_strategy = DESTROY_TYPE::BLOCKED;
696 
697  // if a random value is higher, than the threshold, perform randomwalk instead
698  std::uniform_real_distribution<float> blocked_dist(0, 1);
699  if (blocked_dist(rnd_generator) >= threshold_blocked)
700  {
701  // std::cout << "Running randomwalk. Threshold: " << threshold_blocked << std::endl;
702  destroy_randomwalk(sol);
703  return;
704  }
705 
706  // find all agents whose end is blocked
707  std::vector<int> blocked_agents;
708  for (int i = 0; i < instance.get_num_of_agents(); i++)
709  {
710  const int loc = instance.get_goal_locations()[i];
711  const int reach_time = sol.paths[i].back().interval.t_min;
712  assertm(sol.paths[i].back().location == loc, "The path does not end in the goal location");
713  // skip agents that dont have any delay
714  if (reach_time == instance.get_heuristic_distance(i, instance.get_start_locations()[i]) - 1)
715  {
716  continue;
717  }
718  assertm(reach_time >= instance.get_heuristic_distance(i, instance.get_start_locations()[i]), "Invalid reach time");
719  assertm(constraint_table_initialized, "Constraint table is not initialized.");
720  int from = loc;
721  if (static_cast<int>(sol.paths[i].size()) > 1)
722  {
723  from = std::prev(sol.paths[i].end(), 2)->location;
724  }
725  auto [blocking_agent_1, blocking_agent_2] = constraint_table.get_blocking_agent(from, loc, reach_time - 1);
726  if (blocking_agent_1 != -1 || blocking_agent_2 != -1)
727  {
728  blocked_agents.push_back(i);
729  }
730  }
731 
732  // if no agent is blocked, use the randomwalk
733  if (blocked_agents.empty())
734  {
735  destroy_randomwalk(sol);
736  return;
737  }
738  // std::cout << "Number of blocked_agents: " << blocked_agents.size() << std::endl;
739 
740  // choose a random agent from blocked agents
741  std::uniform_int_distribution<int> dist(0, blocked_agents.size() - 1);
742  int chosen_agent = -1;
743  {
744  int idx = dist(rnd_generator);
745  chosen_agent = blocked_agents[idx];
746  }
747  // std::cout << "Chosen agent: " << chosen_agent << std::endl;
748 
749  // add the agent to the neighborhood
750  std::unordered_set<int> neighborhood = {chosen_agent};
751  int idx = -1;
752  sol.destroyed_paths = {chosen_agent};
753  while (static_cast<int>(sol.destroyed_paths.size()) < settings.destroy_settings.size)
754  {
755  const int loc = instance.get_goal_locations()[chosen_agent];
756  const int min_reach_time = instance.get_heuristic_distance(chosen_agent, instance.get_start_locations()[chosen_agent]);
757  std::vector<int> blocking_agents_new = constraint_table.get_blocking_agents(loc, min_reach_time - 1);
758 
759  // add all new blocking agents to the neighborhood
760  for (auto it : blocking_agents_new)
761  {
762  if (neighborhood.find(it) == neighborhood.end())
763  {
764  neighborhood.insert(it);
765  sol.destroyed_paths.push_back(it);
766  // make sure that the neighborhood size is not exceeded
767  // if (static_cast<int>(sol.destroyed_paths.size()) >= settings.destroy_settings.size)
768  // {
769  // break;
770  // }
771  }
772  }
773 
774  // update index
775  idx++;
776  if (idx >= static_cast<int>(sol.destroyed_paths.size()))
777  {
778  break;
779  }
780 
781  // update chosen agent
782  chosen_agent = sol.destroyed_paths[idx];
783  }
784  // shuffle randomly, only when threshold is low (we need more variance)
785  if (blocked_dist(rnd_generator) >= threshold_blocked)
786  {
787  // std::cout << "Shuffle. Threshold: " << threshold_blocked << std::endl;
788  std::shuffle(sol.destroyed_paths.begin(), sol.destroyed_paths.end(), rnd_generator);
789  }
790  // dont exceed the neighborhood size
791  if (settings.destroy_settings.size < static_cast<int>(sol.destroyed_paths.size()))
792  {
793  sol.destroyed_paths.resize(settings.destroy_settings.size);
794  }
795  assertm(settings.destroy_settings.size == 1 || static_cast<int>(sol.destroyed_paths.size()) > 1,
796  "At least two paths should be destroyed.");
797  sol.feasible = false;
798 }

865  if (constraint_table_initialized)
866  {
867  constraint_table.add_constraints(tp_path, path_idx);
