18  std::vector<int> ret;

27  {
28  continue; // right edge

117  type = split_line[1]; // read type
118  }

120  {
121  try

124  }

129  }

131  {

135  }
136  catch (const std::exception& e)
137  {
138  throw std::runtime_error("Unable to read the number of rows.");
139  }
140  }
141  else if (boost::iequals(split_line[0], "map")) // read the map
142  {
143  while (getline(map_file, line))
144  {
145  for (char i : line)
146  {
147  // treat . as free cell
148  if (i == '.')
149  {

152  // remember index to the location

154  data.push_back(0);
155  }
156  // treat @, T as obstacle
157  else if (i == '@' || i == 'T')
158  {
159  // there is no index to the free location index
160  location_to_free_location_vec.push_back(-1);
161  data.push_back(1);
162  }
163  // report all other symbols as unknown
164  else
165  {

167  }
168  }
169  // check whether the number of cells in this row is consistent with num_of_columns
170  if (static_cast<int>(data.size()) % num_of_cols != 0)
171  {
172  throw std::runtime_error("Inconsistent number of symbols in a row in the map description.");
173  }
174  }
175  // check whether the number of lines is consistent with num_of_rows
176  if (static_cast<int>(data.size()) != num_of_cols * num_of_rows)
177  {
178  throw std::runtime_error("Inconsistent number of rows in the map description.");
179  }
180  }
181  }
182 
183  if (num_of_cols <= 0)
184  {
185  throw std::runtime_error("Invalid number of columns.");
186  }
187  if (num_of_rows <= 0)
188  {
189  throw std::runtime_error("Invalid number of rows.");
190  }
191 
192  assertm(location_to_free_location_vec.size() == data.size(), "Each location must have an index to the free location vector.");
193 
194  // close the map file
195  map_file.close();
196 
197  // assign the size of the map
198  height = num_of_rows;
199  width = num_of_cols;
200  loaded = true;
201 }
