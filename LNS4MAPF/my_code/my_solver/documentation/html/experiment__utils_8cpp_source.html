108 auto what_can_be_skipped(const std::string& experiment_name, const std::vector<std::string>& maps,
109  const std::vector<std::vector<int>>& agent_nums, int runs_per_map, const std::vector<Algorithm>& algorithms)
110  -> std::vector<bool>
111 {
112  assertm(maps.size() == agent_nums.size(), "The number of maps must be the same as the number of agent numbers lists");
113  std::vector<bool> skips;
114  const std::string result_dir_path = get_base_path() + "/experiments/" + experiment_name + "/";
115 
116  // iterate over the maps
117  for (int i = 0; i < static_cast<int>(maps.size()); i++)
118  {
119  const std::string& map_name = maps[i];
120  const std::string map_dir = result_dir_path + map_name;
121  // std::cout << "Map dir: " << map_dir << std::endl;
122  if (!std::filesystem::exists(map_dir))
123  {
124  std::filesystem::create_directory(map_dir);
125  // cant skip files for which the directory did not exist
126  skips.resize(skips.size() + agent_nums[i].size() * runs_per_map * algorithms.size(), false);
127  continue;
128  }
129 
130  skips.reserve(skips.size() + agent_nums[i].size() * runs_per_map * algorithms.size());
131 
132 
133  std::vector<std::string> scenes = get_scene_names(map_name, runs_per_map);
134  assertm(!scenes.empty(), "Did not find any scenes");
135 
136  // iterate over the agent numbers
137  for (int j = 0; j < (int)agent_nums[i].size(); j++)
138  {
139  assertm(i <= (int)agent_nums.size(), "Invalid map index");
140  assertm(j <= (int)agent_nums[i].size(), "Invalid map index");
141  for (int k = 0; k < runs_per_map; k++)
142  {
143  const int idx = k % (int)scenes.size();
144  const std::string& scene_name = scenes[idx];
145  size_t dot_pos = scene_name.find('.');
146  if (dot_pos == std::string::npos)
147  {
148  throw std::runtime_error("Could not find dot in scene name: '" + scene_name + "'");
149  }
150  const std::string scene_name_stripped = scene_name.substr(map_name.size() + 1, dot_pos - map_name.size() - 1);
151 
152  for (const auto& algo : algorithms)
153  {
154  const std::string algorithm_name = algo.get_name();
155  const std::string filename = map_dir + "/" + experiment_name + "_" + std::to_string(agent_nums[i][j]) + "agents_" + map_name +
156  "_" + scene_name_stripped + "_" + algorithm_name + "_" +
157  std::to_string((int)std::floor(k / scenes.size())) + ".json";
158 
159  assertm((int)filename.size() < 260, "Filename must be shorter than 260 characters.");
160  // check whether the file exists
161  if (std::filesystem::exists(filename))
162  {
163  skips.push_back(true);
164  }
165  else
166  {
167  skips.push_back(false);
168  }
169  }
170  }
171  }
172  }
173  return skips;
174 }

178 {
179  // convert the enum to string
180  std::string ret = std::visit([](auto&& arg) -> std::string { return std::string(magic_enum::enum_name(arg)); }, type);
181 
182  // if there are no parameters, return the string
183  if (parameters.empty())
184  {
185  return ret;
186  }
187 
188  // else add the parameters to the string representation
189  for (const auto& param : parameters)
190  {
191  // add the parameter name
192  ret += "_" + param.first + "_";
193 
194  // convert the parameter value to string
195  std::string param_val_str = any_to_str(param.second);
196 
197  ret += param_val_str;
198  }
199  return ret;
