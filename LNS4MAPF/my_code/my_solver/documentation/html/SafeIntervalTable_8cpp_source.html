36 void SafeIntervalTable::add_constraint(const TimePoint& timepoint)
37 {
38  // check the location is valid
39  assertm(instance.get_map_data().is_in(timepoint.location), "Invalid location.");
40 
41  // update unlimited safe interval number
42  if (timepoint.interval.t_max == INT_MAX)
43  {
44  unlimited_safe_intervals--;
45  assertm(unlimited_safe_intervals >= 0, "Invalid unlimited safe interval count.");
46  }
47  else
48  {
49  // update latest constraint end
50  latest_constraint_end = std::max(latest_constraint_end, timepoint.interval.t_max);
51  }
52 
53  int free_location = instance.location_to_free_location(timepoint.location);
54  std::list<TimeInterval>& cur_TI_list = safe_intervals[free_location];
55 
56  // modify intervals that have some intersections with the time range
57  auto it = cur_TI_list.begin();
58  while (it != cur_TI_list.end())
59  {
60  // skip irrelevant time intervals
61  if (it->t_max < timepoint.interval.t_min)
62  {
63  it++;
64  continue;
65  }
66 
67  // check that there are still intervals that could be reduced
68  assertm(it->t_min <= timepoint.interval.t_max, "Can not add an overlapping constraint.");
69 
70  // shorten intervals starting before the range
71  if (it->t_min < timepoint.interval.t_min)
72  {
73  assertm(it->t_max >= timepoint.interval.t_max, "Can not add an overlapping constraint.");
74  const int new_tmax = timepoint.interval.t_min - 1;
75 
76  // create new interval if there is still some time in the safe interval after the constraint
77  if (it->t_max > timepoint.interval.t_max)
78  {
79  cur_TI_list.insert(std::next(it, 1), TimeInterval(timepoint.interval.t_max + 1, it->t_max));
80  it->t_max = new_tmax;
81  break;
82  }
83  const int tmax = it->t_max;
84 
85  it->t_max = new_tmax;
86 
87  // check whether there might be any other affected intervals
88  if (tmax == timepoint.interval.t_max)
89  {
90  break;
91  }
92  it++;
93  }
94  // shorten intervals ending after the range
95  else if (it->t_max > timepoint.interval.t_max)
96  {
97  assertm(it->t_min == timepoint.interval.t_min, "Can not add an overlapping constraint.");
98  it->t_min = timepoint.interval.t_max + 1;
99  break;
100  }
101  // delete intervals that are fully overlapped by the constraint
102  else
103  {
104  const int tmax = it->t_max;
105  it = cur_TI_list.erase(it);
106 
107  // check whether there can be any other affected ontervals
108  if (tmax == timepoint.interval.t_max)
109  {
110  break;
111  }
112  }
113  }
114 }

117 void SafeIntervalTable::remove_constraint(const TimePoint& timepoint)
118 {
119  // check the location is valid
120  assertm(instance.get_map_data().is_in(timepoint.location), "Invalid location.");
121 
122  // update unlimited safe interval number
123  if (timepoint.interval.t_max == INT_MAX)
124  {
125  unlimited_safe_intervals++;
126  assertm(unlimited_safe_intervals <= instance.get_map_data().get_num_free_cells(), "Invalid unlimited safe interval count.");
127 
128  // update latest constraint done in remove constraints
129  }
130 
131  int free_location = instance.location_to_free_location(timepoint.location);
132  std::list<TimeInterval>& cur_TI_list = safe_intervals[free_location];
133 
134  // check whether there are any safe intervals
135  if (cur_TI_list.empty())
136  {
137  cur_TI_list.push_back(timepoint.interval);
138  return;
139  }
140 
141  // firstly handle constraints after the last safe interval (there is no safe interval after the constraint)
142  assertm(timepoint.interval.t_min != cur_TI_list.back().t_max, "Constraint start overlaps with last safe interval end.");
143 
144  if (timepoint.interval.t_min > cur_TI_list.back().t_max)
145  {
146  // if the last safe interval precedes directly, extend it
147  if (cur_TI_list.back().t_max == timepoint.interval.t_min - 1)
148  {
149  cur_TI_list.back().t_max = timepoint.interval.t_max;
150  return;
151  }
152  // otherwise add a new safe interval
153  cur_TI_list.push_back(timepoint.interval);
154  return;
155  }
156 
157  // find interval after the constraint
158  auto it = cur_TI_list.begin();
159  while (it != cur_TI_list.end())
160  {
161  assertm(!overlap(*it, timepoint.interval), "Constraint interval can not have any overlap with safe interval.");
162 
163  // find the interval after the constraint
164  if (it->t_min > timepoint.interval.t_max)
165  {
166  // check whether there was an interval before the constraint
167  if (it != cur_TI_list.begin())
168  {
169  // check whether the previous timeinterval can be extended
170  const auto prev = std::prev(it, 1);
171  if (prev->t_max == timepoint.interval.t_min - 1)
172  {
173  // check whether prev should be merged with next interval
174  if (timepoint.interval.t_max != INT_MAX && it->t_min == timepoint.interval.t_max + 1)
175  {
176  prev->t_max = it->t_max;
177  it = cur_TI_list.erase(it);
178  }
179  else // extend prev
180  {
181  prev->t_max = timepoint.interval.t_max;
182  }
183  return;
184  }
185  }
186 
187  // previous can not be extended, check whether next can be extended
188  if (timepoint.interval.t_max != INT_MAX && it->t_min == timepoint.interval.t_max + 1)
189  {
190  it->t_min = timepoint.interval.t_min;
191  return;
192  }
193 
194  // no interval can be extended, therefore insert a new interval
195  cur_TI_list.insert(it, timepoint.interval);
196  return;
197  }
198  it++;
199  }
200  throw std::runtime_error("Did not remove any constraint.");
201 }

203 void SafeIntervalTable::add_constraints(const TimePointPath& path)
204 {
205  for (int i = 0; i < (int)path.size(); i++)
206  {
207  // add the point constraint
208  add_constraint(path[i]);
209 
210  // add the edge constraint
211  if (i > 0) // skip the first point as there is no edge
212  {
213  // the time when the edge is used is the start time of the later interval
214  edge_constraint_table.add(path[i - 1].location, path[i].location, path[i].interval.t_min);
215  }
216  }

305 auto SafeIntervalTable::get_safe_intervals(int location, const TimeInterval& time_interval) const
306  -> std::pair<std::list<TimeInterval>::const_iterator, std::list<TimeInterval>::const_iterator>
307 {
308  // this implementation does not check edge collisions, so they must be handled elsewhere.
309  assertm(instance.get_map_data().is_in(location), "Invalid location.");
310  int free_location = instance.location_to_free_location(location);
311  assertm(free_location < static_cast<int>(safe_intervals.size()), "Safe intervals not precomputed for this location.");
312 
313  // iterate over the list to find the first and last safe interval
314  auto start = safe_intervals[free_location].cend();
315  auto end = safe_intervals[free_location].cend();
316  for (auto it = safe_intervals[free_location].cbegin(); it != safe_intervals[free_location].cend(); it++)
317  {
318  // std::cout << "Interval: " << it->t_min << " " << it->t_max << std::endl;
319  // check whether the interval is inside the given time range
320  if (it->t_max >= time_interval.t_min && it->t_min <= time_interval.t_max)
321  {
322  // remember the first interval in the range
323  if (start == safe_intervals[free_location].end())
324  {
325  start = it;
326  }
327  }
328  // if the interval is not inside the time range, check whether already found some intervals
329  else if (start != safe_intervals[free_location].cend())
330  {
331  // if already found some intervals, return the found safe intervals
332  end = it;
333  return std::make_pair(start, end);
334  }
335  }
336  return std::make_pair(start, end);
337 }
